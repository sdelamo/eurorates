18 package groovycalamari.eurorates;
19 
20 import io.micronaut.http.MediaType;
21 import io.micronaut.http.annotation.Consumes;
22 import io.micronaut.http.annotation.Get;
23 import io.micronaut.http.annotation.Produces;
24 import io.micronaut.http.client.annotation.Client;
25 import io.micronaut.retry.annotation.Retryable;
26 import io.micronaut.validation.Validated;
27 import io.reactivex.Single;
28 
29 @Validated
30 @Client(
31         value = "${" + EuroRatesConfiguration.PREFIX + ".url:`" + EuroRatesConfiguration.HOST_LIVE + "`}",
32         configuration = EuroRatesConfiguration.class
33 )
34 @Retryable(
35         attempts = "${" + EuroRatesConfiguration.PREFIX + ".retry.attempts:0}",
36         delay = "${" + EuroRatesConfiguration.PREFIX + ".retry.delay:5s}")
37 @Consumes(MediaType.APPLICATION_XML)
38 @Produces(MediaType.APPLICATION_XML)
39 public interface EuroRatesClient extends EuroRatesApi {
40 
41     String PATH_CURRENT = "/stats/eurofxref/eurofxref-daily.xml";
42     String PATH_HISTORY = "/stats/eurofxref/eurofxref-hist.xml";
43     String PATH_90_DAYS = "/stats/eurofxref/eurofxref-hist-90d.xml";
44 
45     @Override
46     @Get(PATH_CURRENT)
47     Single<GesmesEnvelope> currentReferenceRates();
48 
49     @Override
50     @Get(PATH_HISTORY)
51     Single<GesmesEnvelope> historicalReferenceRates();
52 
53     @Override
54     @Get(PATH_90_DAYS)
55     Single<GesmesEnvelope> last90DaysReferenceRates();
56 }
